    USER {
        string   username PK "key in users.json"
        string   password_hash "bcrypt"
        string   email
        string   role "admin | user"
        int      tokens "balance"
        datetime created_at
    }

    ANALYSIS_LOG {
        string   ticket_id PK "8-char id (JWT ticket)"
        string   username FK
        string   cui FK "nullable"
        string   company_name "nullable"
        string   services "serp|anaf|intel|berc|ai"
        int      cost "tokens deducted"
        int      remaining_tokens
        datetime timestamp
    }

    FEEDBACK {
        string   id PK "8-char id"
        string   cui FK "bucket key in feedback.json"
        string   username FK
        int      rating "1-5"
        string   comment "nullable"
        datetime created_at
    }

    ALERT {
        string   username FK "bucket key in alerts.json"
        string   cui FK "unique per user"
        string   company_name
        int      baseline_score
        string   baseline_band
        int      last_score
        string   last_band
        datetime created_at
    }

    SCORE_SNAPSHOT {
        string   cui FK "bucket key in snapshots.json"
        string   company_name
        int      score "0-100"
        string   band "high_risk|caution|healthy"
        datetime timestamp
    }

    COMPANY {
        string cui PK "fiscal code (derived hub, not a stored table)"
        string company_name "denormalized on each record"
    }
